  const restaurant = useSelector(get('restaurant'));

  if (!restaurant) {
    return (
      <p>Loading......</p>

  function handleChange({ name, value }) {
    dispatch(changeReviewField({ name, value }));
  }

  return (

      <RestaurantDetail restaurant={restaurant} />
      <ReviewForm onChange={handleChange} />
    </>
  );
}
