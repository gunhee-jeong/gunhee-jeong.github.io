# 1. 설명
문자와 문자열이 주어졌을때, solution 함수는 주어진 문자열에서 주어진 문자가 나타나는 첫번째 위치를 반환합니다.  
만약 문자열에 해당 문자가 여러번 나타나면, 첫번째로 나타나는 위치를 반환해야 합니다.  
만약 문자가 문자열에 존재하지 않는다면, -1 을 반환해야 합니다.  
indexOf 함수를 사용하지 않고 구현하세요.  

[주어진 코드]  

```js
const char = 'a';
const string = "I am a hacker";

function solution(char, string) {
  // 코드를 작성해주세요.
}

solution(char, string); // 2
```

# 2. 나의 풀이
```js
const char = 'a';
const string = "I am a hacker";

function solution(char, string) {
  let arr = string.split(' ');
  let answer;
  
  arr.forEach((aEle, aInd) => {
    if (aEle === char) {
      return answer = aInd;
    } 
    else if (answer) return true;
    else answer = -1;
  });
  
  return answer;
}

solution(char, string); // 2
```

# 3. 풀이

```js
const char = 'a';
const string = "I am a hacker";

function solution(char, string) {
  let start = string.split(' ');

  for(let i = 0; i < start.length; i++) {
    if (char === start[i]) {
      return i
    }
  }
  return -1
}

solution(char, string);
```
