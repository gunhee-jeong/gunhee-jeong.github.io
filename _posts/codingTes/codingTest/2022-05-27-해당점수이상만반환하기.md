# 2. 나의 풀이

```js
const students = [
  {
    name: 'A',
    score: 45
  }, 
  {
    name: 'B',
    score: 80
  }, 
  {
    name: 'C',
    score: 90
  }, 
  {
    name: 'D',
    score: 66
  }, 
  {
    name: 'E',
    score: 88
  }, 
];

function solution (students) {
  let list = students.filter(aEle => {
    if (50 < aEle.score) return true;
  });
  
  let answer = list.map(bEle => bEle.score).sort().join();
  
  return answer;
}

solution(students); // '66,80,88,90'
```
